## 🎯 概述

**HMC5883L** 是 Honeywell 生产的经典 3 轴磁力计传感器，采用各向异性磁阻 (AMR) 技术，广泛应用于电子罗盘、姿态检测、磁场测量等应用。虽然已停产 (替代型号 QMC5883L)，但因其优秀性能和低成本，仍有大量应用。

**核心特性:**
- ✨ **3 轴测量:** X/Y/Z三轴磁场强度测量
- ✨ **高灵敏度:** 最低 1mG 分辨率
- ✨ **宽量程:** ±1.3Ga 到 ±8.1Ga 可选
- ✨ **低功耗:** 100μA@10Hz 测量
- ✨ **I2C 接口:** 标准 400kHz I2C
- ✨ **小封装:** 3.0 × 3.0 × 0.9 mm LCC

**典型应用:** 电子罗盘、无人机航向、机器人导航、金属探测、地磁测量、姿态参考

### 典型电路

```
         3.3V
           │
      ┌────┴────┐
      │         │
     ┌┴┐       ┌┴┐
     │ │ 100nF │ │ 4.7kΩ
     │ │       │ │ (上拉)
     └┬┘       └┬┘
      │         │
      ├────┬────┼─── SCL (PB6)
      │    │    │
      │   GND   ├─── SDA (PB7)
      │         │
     ┌┴───────────┴┐
     │   HMC5883L  │
     │             │
     │ DRDY → PA0  │  (可选中断)
     └─────────────┘
```

**设计要点:**
1. **去耦电容:** 100nF 陶瓷电容，尽量靠近 VDD 引脚
2. **I2C 上拉:** 4.7kΩ (3.3V 系统)
3. **PCB 布局:** 
   - 远离电机、变压器等磁场源
   - 避免大电流走线
   - 传感器居中放置，周围无铜皮
4. **安装方向:** 标记 X/Y 轴方向，便于软件补偿

## 💻 软件驱动 (STM32 HAL)

### 寄存器定义

```c
#ifndef __HMC5883L_H
#define __HMC5883L_H

#include "main.h"

// HMC5883L I2C 地址
#define HMC5883L_ADDR       0x1E

// 寄存器地址
#define HMC5883L_CONFIG_A   0x00
#define HMC5883L_CONFIG_B   0x01
#define HMC5883L_MODE       0x02
#define HMC5883L_DATA_X_H   0x03
#define HMC5883L_DATA_X_L   0x04
#define HMC5883L_DATA_Z_H   0x05
#define HMC5883L_DATA_Z_L   0x06
#define HMC5883L_DATA_Y_H   0x07
#define HMC5883L_DATA_Y_L   0x08
#define HMC5883L_STATUS     0x09
#define HMC5883L_ID_A       0x0A
#define HMC5883L_ID_B       0x0B
#define HMC5883L_ID_C       0x0C

// 芯片 ID
#define HMC5883L_ID "H43"

// 量程配置 (CONFIG_B)
typedef enum {
    HMC5883L_RANGE_1_3GA  = 0x20,  // ±1.3Ga, 0.73 mG/LSB
    HMC5883L_RANGE_1_9GA  = 0x40,  // ±1.9Ga, 0.92 mG/LSB
    HMC5883L_RANGE_2_5GA  = 0x60,  // ±2.5Ga, 1.22 mG/LSB
    HMC5883L_RANGE_4_0GA  = 0x80,  // ±4.0Ga, 1.52 mG/LSB
    HMC5883L_RANGE_4_7GA  = 0xA0,  // ±4.7Ga, 2.27 mG/LSB
    HMC5883L_RANGE_5_6GA  = 0xC0,  // ±5.6Ga, 2.56 mG/LSB
    HMC5883L_RANGE_8_1GA  = 0xE0   // ±8.1Ga, 4.35 mG/LSB
} HMC5883L_Range_t;

// 工作模式
typedef enum {
    HMC5883L_MODE_CONTINUOUS = 0x00,  // 连续测量
    HMC5883L_MODE_SINGLE     = 0x01,  // 单次测量
    HMC5883L_MODE_IDLE       = 0x02   // 空闲模式
} HMC5883L_Mode_t;

// 数据输出结构
typedef struct {
    int16_t raw_x, raw_y, raw_z;    // 原始数据
    float mag_x, mag_y, mag_z;      // 磁场强度 (mG)
    float heading;                   // 航向角 (度)
    HAL_StatusTypeDef last_status;
} HMC5883L_Data_t;

// 句柄结构
typedef struct {
    I2C_HandleTypeDef *hi2c;
    uint8_t address;
    HMC5883L_Range_t range;
    float gain;  // 增益 (mG/LSB)
} HMC5883L_Handle_t;

#endif
```

### 初始化函数

```c
/**
 * @brief HMC5883L 初始化
 * @param handle: 句柄
 * @param hi2c: I2C 句柄
 * @param address: I2C 地址
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef HMC5883L_Init(HMC5883L_Handle_t *handle,
                                 I2C_HandleTypeDef *hi2c,
                                 uint8_t address) {
    uint8_t chip_id[3];
    uint8_t reg = 0;
    
    handle->hi2c = hi2c;
    handle->address = address;
    handle->range = HMC5883L_RANGE_1_3GA;
    handle->gain = 0.73f;  // mG/LSB
    
    // 1. 读取芯片 ID
    HAL_I2C_Mem_Read(hi2c, address << 1, HMC5883L_ID_A,
                     I2C_MEMADD_SIZE_8BIT, chip_id, 3, 100);
    
    // 验证 ID ("H43")
    if (chip_id[0] != 'H' || chip_id[1] != '4' || chip_id[2] != '3') {
        return HAL_ERROR;
    }
    
    // 2. 配置 CONFIG_A (采样率 15Hz, 正常测量)
    reg = 0x70;  // 15Hz, 2 个样本平均
    HAL_I2C_Mem_Write(hi2c, address << 1, HMC5883L_CONFIG_A,
                      I2C_MEMADD_SIZE_8BIT, &reg, 1, 100);
    
    // 3. 配置 CONFIG_B (量程 ±1.3Ga)
    reg = handle->range;
    HAL_I2C_Mem_Write(hi2c, address << 1, HMC5883L_CONFIG_B,
                      I2C_MEMADD_SIZE_8BIT, &reg, 1, 100);
    
    // 4. 配置模式 (连续测量)
    reg = HMC5883L_MODE_CONTINUOUS;
    HAL_I2C_Mem_Write(hi2c, address << 1, HMC5883L_MODE,
                      I2C_MEMADD_SIZE_8BIT, &reg, 1, 100);
    
    HAL_Delay(100);  // 等待测量稳定
    
    return HAL_OK;
}

/**
 * @brief 设置量程
 * @param handle: 句柄
 * @param range: 量程选择
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef HMC5883L_SetRange(HMC5883L_Handle_t *handle,
                                     HMC5883L_Range_t range) {
    uint8_t reg = range;
    HAL_StatusTypeDef status = HAL_I2C_Mem_Write(handle->hi2c,
                                                  handle->address << 1,
                                                  HMC5883L_CONFIG_B,
                                                  I2C_MEMADD_SIZE_8BIT,
                                                  &reg, 1, 100);
    if (status == HAL_OK) {
        handle->range = range;
        // 更新增益
        float gains[] = {0.73f, 0.92f, 1.22f, 1.52f, 2.27f, 2.56f, 4.35f};
        handle->gain = gains[range >> 5];
    }
    return status;
}
```

### 数据读取函数

```c
/**
 * @brief 读取磁力计数据
 * @param handle: 句柄
 * @param data: 数据输出
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef HMC5883L_ReadData(HMC5883L_Handle_t *handle,
                                     HMC5883L_Data_t *data) {
    uint8_t buffer[6];
    
    // 读取 6 字节数据 (X, Z, Y 顺序)
    data->last_status = HAL_I2C_Mem_Read(handle->hi2c,
                                          handle->address << 1,
                                          HMC5883L_DATA_X_H,
                                          I2C_MEMADD_SIZE_8BIT,
                                          buffer, 6, 100);
    
    if (data->last_status != HAL_OK) {
        return data->last_status;
    }
    
    // 解析原始数据
    data->raw_x = (int16_t)((buffer[0] << 8) | buffer[1]);
    data->raw_z = (int16_t)((buffer[2] << 8) | buffer[3]);
    data->raw_y = (int16_t)((buffer[4] << 8) | buffer[5]);
    
    // 转换为磁场强度 (mG)
    data->mag_x = data->raw_x * handle->gain;
    data->mag_y = data->raw_y * handle->gain;
    data->mag_z = data->raw_z * handle->gain;
    
    // 计算航向角 (假设水平放置)
    data->heading = atan2f(-data->mag_y, data->mag_x) * 180.0f / M_PI;
    if (data->heading < 0) {
        data->heading += 360.0f;
    }
    
    return HAL_OK;
}

/**
 * @brief 单次测量模式
 * @param handle: 句柄
 * @param data: 数据输出
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef HMC5883L_SingleMeasure(HMC5883L_Handle_t *handle,
                                          HMC5883L_Data_t *data) {
    uint8_t mode = HMC5883L_MODE_SINGLE;
    
    // 设置为单次模式
    HAL_I2C_Mem_Write(handle->hi2c, handle->address << 1,
                      HMC5883L_MODE, I2C_MEMADD_SIZE_8BIT,
                      &mode, 1, 100);
    
    // 等待测量完成 (约 67ms)
    HAL_Delay(100);
    
    // 读取数据
    return HMC5883L_ReadData(handle, data);
}
```

### 校准函数

```c
// 校准数据结构
typedef struct {
    float offset_x, offset_y, offset_z;
    float scale_x, scale_y, scale_z;
} HMC5883L_Calib_t;

/**
 * @brief 应用校准数据
 * @param data: 原始数据
 * @param calib: 校准参数
 */
void HMC5883L_ApplyCalibration(HMC5883L_Data_t *data,
                                HMC5883L_Calib_t *calib) {
    // 应用零偏
    data->mag_x = (data->mag_x - calib->offset_x) * calib->scale_x;
    data->mag_y = (data->mag_y - calib->offset_y) * calib->scale_y;
    data->mag_z = (data->mag_z - calib->offset_z) * calib->scale_z;
    
    // 重新计算航向
    data->heading = atan2f(-data->mag_y, data->mag_x) * 180.0f / M_PI;
    if (data->heading < 0) data->heading += 360.0f;
}

/**
 * @brief 简易校准 (画 8 字法)
 * @param handle: 句柄
 * @param calib: 校准输出
 * @param samples: 采样数
 */
void HMC5883L_Calibrate(HMC5883L_Handle_t *handle,
                         HMC5883L_Calib_t *calib,
                         int samples) {
    HMC5883L_Data_t data;
    float min_x = -1000, max_x = 1000;
    float min_y = -1000, max_y = 1000;
    float min_z = -1000, max_z = 1000;
    
    printf("开始校准，请在空中画 8 字...\r\n");
    
    for (int i = 0; i < samples; i++) {
        HMC5883L_ReadData(handle, &data);
        
        if (data.mag_x < min_x) min_x = data.mag_x;
        if (data.mag_x > max_x) max_x = data.mag_x;
        if (data.mag_y < min_y) min_y = data.mag_y;
        if (data.mag_y > max_y) max_y = data.mag_y;
        if (data.mag_z < min_z) min_z = data.mag_z;
        if (data.mag_z > max_z) max_z = data.mag_z;
        
        printf("校准中：%d/%d\r", i + 1, samples);
        HAL_Delay(100);
    }
    printf("\r\n");
    
    // 计算零偏
    calib->offset_x = (max_x + min_x) / 2.0f;
    calib->offset_y = (max_y + min_y) / 2.0f;
    calib->offset_z = (max_z + min_z) / 2.0f;
    
    // 计算比例因子
    float avg_range = ((max_x - min_x) + (max_y - min_y) + (max_z - min_z)) / 6.0f;
    calib->scale_x = avg_range / ((max_x - min_x) / 2.0f);
    calib->scale_y = avg_range / ((max_y - min_y) / 2.0f);
    calib->scale_z = avg_range / ((max_z - min_z) / 2.0f);
    
    printf("校准完成:\r\n");
    printf("Offset: %.2f, %.2f, %.2f\r\n",
           calib->offset_x, calib->offset_y, calib->offset_z);
    printf("Scale: %.3f, %.3f, %.3f\r\n",
           calib->scale_x, calib->scale_y, calib->scale_z);
}
```

## 🔧 调试技巧

### 常见问题

| 问题 | 解决方案 |
|------|----------|
| I2C 无响应 | 检查供电，确认 I2C 地址 (0x1E) |
| 数据全为 0 | 检查模式配置，确保非空闲模式 |
| 航向漂移 | 进行校准，远离磁场干扰 |
| 数据跳变 | 增加滤波，降低采样率 |

### 航向计算 (带倾角补偿)

```c
/**
 * @brief 计算倾角补偿航向
 * @param mag: 磁力计数据
 * @param accel: 加速度计数据 (需要先归一化)
 * @return 补偿后航向 (度)
 */
float HMC5883L_CalculateTiltCompensatedHeading(HMC5883L_Data_t *mag,
                                                float accel_x,
                                                float accel_y,
                                                float accel_z) {
    // 计算俯仰和横滚角
    float pitch = atan2f(-accel_x, sqrtf(accel_y * accel_y + accel_z * accel_z));
    float roll = atan2f(accel_y, accel_z);
    
    // 倾角补偿
    float xh = mag->mag_x * cosf(pitch) + mag->mag_z * sinf(pitch);
    float yh = mag->mag_x * sinf(roll) * sinf(pitch) +
               mag->mag_y * cosf(roll) -
               mag->mag_z * sinf(roll) * cosf(pitch);
    
    float heading = atan2f(-yh, xh) * 180.0f / M_PI;
    if (heading < 0) heading += 360.0f;
    
    return heading;
}
```
