# BME280 温湿度气压传感器

> **一句话描述:** Bosch 经典三合一环境传感器，集成温度 + 湿度 + 气压，I2C/SPI 双接口，高精度低功耗  
> **标签:** `#sensor/temperature` `#sensor/humidity` `#sensor/pressure` `#protocol/i2c` `#protocol/spi` `#vendor/bosch`  
> **状态:** ✅ 完成  
> **最后更新:** 2026-03-11

---

## 🎯 概述

**BME280** 是 Bosch Sensortec 生产的三合一环境传感器，在 BMP280 基础上增加了湿度测量，成为智能家居、气象站、可穿戴设备的热门选择。

**核心特性:**
- ✨ **三合一:** 温度 + 湿度 + 气压单芯片集成
- ✨ **高精度:** ±1hPa 气压，±3% RH 湿度，±1.0°C 温度
- ✨ **低功耗:** 测量模式 3.6μA@1Hz，睡眠 0.1μA
- ✨ **快速响应:** 气压 5.5ms，湿度 1s
- ✨ **双接口:** I2C (3.4MHz) 或 SPI (3.4MHz)
- ✨ **宽范围:** 300-1100hPa，-40~85°C，0-100% RH

**典型应用:** 气象站、室内空气质量监测、GPS 高度辅助、可穿戴设备、智能家居、无人机

## 🔌 硬件设计

### 引脚定义

```
        BME280 顶视图
        
    VDD  1 ○ ○ 2 GND
         
    GND  3 ○ ○ 4 SCK
         
    CSB  5 ○ ○ 6 SDI
         
    GND  7 ○ ○ 8 SDO
```

| 引脚 | 名称 | 类型 | I2C 模式 | SPI 模式 |
|------|------|------|----------|----------|
| 1 | VDD | P | 电源 | 电源 |
| 2 | GND | P | 地 | 地 |
| 3 | GND | P | 地 | 地 |
| 4 | SCK | I | SCL | SPI 时钟 |
| 5 | CSB | I | 接 VDD(0x77) 或 GND(0x76) | SPI 片选 |
| 6 | SDI | I/O | SDA | SPI MOSI |
| 7 | GND | P | 地 | 地 |
| 8 | SDO | O | 未使用 | SPI MISO |

### I2C 地址选择

| CSB 引脚 | I2C 地址 |
|----------|----------|
| 接 GND | 0x76 |
| 接 VDD | 0x77 |

### 典型电路 (I2C 模式)

```
         3.3V
           │
      ┌────┴────┐
      │         │
     ┌┴┐       ┌┴┐
     │ │ 1μF   │ │ 4.7kΩ
     │ │       │ │ (上拉)
     └┬┘       └┬┘
      │         │
      ├────┬────┼─── SCL
      │    │    │
      │   GND   ├─── SDA
      │         │
     ┌┴───────────┴┐
     │    BME280   │
     │             │
     │ CSB → GND   │  (地址 0x76)
     └─────────────┘
```

**设计要点:**
1. **去耦电容:** 1μF 陶瓷电容，靠近 VDD 引脚
2. **I2C 上拉:** 4.7kΩ (3.3V)，可降至 2.2kΩ 提高速度
3. **PCB 布局:** 
   - 底部焊盘接地
   - 开槽隔离热源
   - 顶部开窗通气
4. **气压孔:** 确保传感器上方有通气孔

## 💻 软件驱动 (STM32 HAL)

### 补偿参数结构

```c
#ifndef __BME280_H
#define __BME280_H

#include "main.h"

// I2C 地址
#define BME280_ADDR_0   0x76
#define BME280_ADDR_1   0x77

// 寄存器地址
#define BME280_REG_CALIB_T1     0x88
#define BME280_REG_CALIB_T2     0x8A
#define BME280_REG_CALIB_T3     0x8C
#define BME280_REG_CALIB_P1     0x8E
#define BME280_REG_CALIB_P2     0x90
#define BME280_REG_CALIB_H1     0xA1
#define BME280_REG_CALIB_H2     0xE1
#define BME280_REG_CHIPID       0xD0
#define BME280_REG_CTRL_HUM     0xF2
#define BME280_REG_STATUS       0xF3
#define BME280_REG_CTRL_MEAS    0xF4
#define BME280_REG_CONFIG       0xF5
#define BME280_REG_PRESSURE     0xF7
#define BME280_REG_TEMPERATURE  0xFA
#define BME280_REG_HUMIDITY     0xFD

// 芯片 ID
#define BME280_CHIP_ID          0x60

// oversampling 设置
#define BME280_OVERSAMPLING_NONE    0x00
#define BME280_OVERSAMPLING_X1      0x01
#define BME280_OVERSAMPLING_X2      0x02
#define BME280_OVERSAMPLING_X4      0x03
#define BME280_OVERSAMPLING_X8      0x04
#define BME280_OVERSAMPLING_X16     0x05

// 模式设置
#define BME280_MODE_SLEEP       0x00
#define BME280_MODE_FORCED      0x01
#define BME280_MODE_NORMAL      0x03

// 补偿参数结构 (工厂校准)
typedef struct {
    uint16_t T1;
    int16_t  T2;
    int16_t  T3;
    uint16_t P1;
    int16_t  P2;
    int16_t  P3;
    int16_t  P4;
    int16_t  P5;
    int16_t  P6;
    int16_t  P7;
    int16_t  P8;
    int16_t  P9;
    uint8_t  H1;
    int16_t  H2;
    uint8_t  H3;
    int16_t  H4;
    int16_t  H5;
    int8_t   H6;
} BME280_CalibData_t;

// 数据输出结构
typedef struct {
    float temperature;  // °C
    float pressure;     // hPa
    float humidity;     // %RH
    float altitude;     // m (计算值)
} BME280_Data_t;

// 句柄结构
typedef struct {
    I2C_HandleTypeDef *hi2c;
    uint8_t address;
    BME280_CalibData_t calib;
} BME280_Handle_t;

#endif
```

### 读取校准参数

```c
/**
 * @brief 读取工厂校准参数
 * @param handle: 句柄
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef BME280_ReadCalibration(BME280_Handle_t *handle) {
    uint8_t buffer[26];
    BME280_CalibData_t *c = &handle->calib;
    
    // 读取温度补偿参数
    HAL_I2C_Mem_Read(handle->hi2c, handle->address << 1,
                     BME280_REG_CALIB_T1,
                     I2C_MEMADD_SIZE_8BIT, buffer, 24, 100);
    
    c->T1 = (uint16_t)(buffer[0] | (buffer[1] << 8));
    c->T2 = (int16_t)(buffer[2] | (buffer[3] << 8));
    c->T3 = (int16_t)(buffer[4] | (buffer[5] << 8));
    
    // 读取气压补偿参数
    c->P1 = (uint16_t)(buffer[6] | (buffer[7] << 8));
    c->P2 = (int16_t)(buffer[8] | (buffer[9] << 8));
    c->P3 = (int16_t)(buffer[10] | (buffer[11] << 8));
    c->P4 = (int16_t)(buffer[12] | (buffer[13] << 8));
    c->P5 = (int16_t)(buffer[14] | (buffer[15] << 8));
    c->P6 = (int16_t)(buffer[16] | (buffer[17] << 8));
    c->P7 = (int16_t)(buffer[18] | (buffer[19] << 8));
    c->P8 = (int16_t)(buffer[20] | (buffer[21] << 8));
    c->P9 = (int16_t)(buffer[22] | (buffer[23] << 8));
    
    // 读取湿度补偿参数
    HAL_I2C_Mem_Read(handle->hi2c, handle->address << 1,
                     BME280_REG_CALIB_H1,
                     I2C_MEMADD_SIZE_8BIT, &c->H1, 1, 100);
    
    HAL_I2C_Mem_Read(handle->hi2c, handle->address << 1,
                     BME280_REG_CALIB_H2,
                     I2C_MEMADD_SIZE_8BIT, buffer, 7, 100);
    
    c->H2 = (int16_t)(buffer[0] | (buffer[1] << 8));
    c->H3 = buffer[2];
    c->H4 = (buffer[3] << 4) | (buffer[4] & 0x0F);
    c->H5 = (buffer[5] << 4) | ((buffer[4] >> 4) & 0x0F);
    c->H6 = (int8_t)buffer[6];
    
    return HAL_OK;
}
```

### 初始化函数

```c
/**
 * @brief BME280 初始化
 * @param handle: 句柄
 * @param hi2c: I2C 句柄
 * @param address: I2C 地址
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef BME280_Init(BME280_Handle_t *handle,
                               I2C_HandleTypeDef *hi2c,
                               uint8_t address) {
    uint8_t chip_id = 0;
    
    handle->hi2c = hi2c;
    handle->address = address;
    
    // 1. 读取芯片 ID
    HAL_I2C_Mem_Read(hi2c, address << 1, BME280_REG_CHIPID,
                     I2C_MEMADD_SIZE_8BIT, &chip_id, 1, 100);
    
    if (chip_id != BME280_CHIP_ID) {
        return HAL_ERROR;
    }
    
    // 2. 读取校准参数
    if (BME280_ReadCalibration(handle) != HAL_OK) {
        return HAL_ERROR;
    }
    
    // 3. 进入睡眠模式 (确保已知状态)
    uint8_t ctrl_meas = BME280_MODE_SLEEP << 5;
    HAL_I2C_Mem_Write(hi2c, address << 1, BME280_REG_CTRL_MEAS,
                      I2C_MEMADD_SIZE_8BIT, &ctrl_meas, 1, 100);
    HAL_Delay(10);
    
    // 4. 配置湿度 oversampling (x1)
    uint8_t ctrl_hum = BME280_OVERSAMPLING_X1;
    HAL_I2C_Mem_Write(hi2c, address << 1, BME280_REG_CTRL_HUM,
                      I2C_MEMADD_SIZE_8BIT, &ctrl_hum, 1, 100);
    
    // 5. 配置正常模式，温度/气压 oversampling x1
    ctrl_meas = (BME280_OVERSAMPLING_X1 << 5) |  // T oversampling
                (BME280_OVERSAMPLING_X1 << 2) |  // P oversampling
                (BME280_MODE_NORMAL << 0);        // Mode
    HAL_I2C_Mem_Write(hi2c, address << 1, BME280_REG_CTRL_MEAS,
                      I2C_MEMADD_SIZE_8BIT, &ctrl_meas, 1, 100);
    
    // 6. 配置 IIR 滤波器 (系数 4)
    uint8_t config = 0x03;
    HAL_I2C_Mem_Write(hi2c, address << 1, BME280_REG_CONFIG,
                      I2C_MEMADD_SIZE_8BIT, &config, 1, 100);
    
    return HAL_OK;
}
```

### 温度补偿计算

```c
/**
 * @brief 补偿温度 (返回 fine_resolution，用于气压/湿度补偿)
 * @param handle: 句柄
 * @param raw_temp: 原始温度值
 * @return 补偿后温度 (°C)
 */
int32_t BME280_CompensateTemperature(BME20948_Handle_t *handle,
                                      int32_t raw_temp) {
    int32_t var1, var2;
    BME280_CalibData_t *c = &handle->calib;
    
    var1 = ((((raw_temp >> 3) - ((int32_t)c->T1 << 1))) *
            ((int32_t)c->T2)) >> 11;
    
    var2 = (((((raw_temp >> 4) - ((int32_t)c->T1)) *
              ((raw_temp >> 4) - ((int32_t)c->T1))) >> 12) *
            ((int32_t)c->T3)) >> 14;
    
    // 保存 fine_resolution 用于后续计算
    int32_t t_fine = var1 + var2;
    
    // 返回温度 (°C × 100)
    return (t_fine * 5 + 128) >> 8;
}
```

### 气压补偿计算

```c
/**
 * @brief 补偿气压
 * @param handle: 句柄
 * @param raw_pressure: 原始气压值
 * @param t_fine: 温度 fine_resolution
 * @return 补偿后气压 (Pa)
 */
uint32_t BME280_CompensatePressure(BME280_Handle_t *handle,
                                    int32_t raw_pressure,
                                    int32_t t_fine) {
    int64_t var1, var2, p;
    BME280_CalibData_t *c = &handle->calib;
    
    var1 = ((int64_t)t_fine) - 128000;
    var2 = var1 * var1 * (int64_t)c->P6;
    var2 = var2 + ((var1 * (int64_t)c->P5) << 17);
    var2 = var2 + (((int64_t)c->P4) << 35);
    var1 = ((var1 * var1 * (int64_t)c->P3) >> 8) +
           ((var1 * (int64_t)c->P2) << 12);
    var1 = (((((int64_t)1) << 47) + var1)) * ((int64_t)c->P1) >> 33;
    
    if (var1 == 0) return 0;
    
    p = 1048576 - raw_pressure;
    p = (((p << 31) - var2) * 3125) / var1;
    var1 = (((int64_t)c->P9) * (p >> 13) * (p >> 13)) >> 25;
    var2 = (((int64_t)c->P8) * p) >> 19;
    
    p = ((p + var1 + var2) >> 8) + (((int64_t)c->P7) << 4);
    
    return (uint32_t)(p >> 8);
}
```

### 湿度补偿计算

```c
/**
 * @brief 补偿湿度
 * @param handle: 句柄
 * @param raw_humidity: 原始湿度值
 * @param t_fine: 温度 fine_resolution
 * @return 补偿后湿度 (%RH × 1000)
 */
uint32_t BME280_CompensateHumidity(BME280_Handle_t *handle,
                                    int32_t raw_humidity,
                                    int32_t t_fine) {
    int32_t v_x1_u32r;
    BME280_CalibData_t *c = &handle->calib;
    
    v_x1_u32r = (t_fine - ((int32_t)76800));
    v_x1_u32r = (((((raw_humidity << 14) - (((int32_t)c->H4) << 20) -
                    (((int32_t)c->H5) * v_x1_u32r)) + ((int32_t)16384)) >> 15) *
                 (((((((v_x1_u32r * ((int32_t)c->H6)) >> 10) *
                      (((v_x1_u32r * ((int32_t)c->H3)) >> 11) + ((int32_t)32768))) >> 10) +
                    ((int32_t)2097152)) * ((int32_t)c->H2) + 8192) >> 14));
    
    v_x1_u32r = (v_x1_u32r - (((((v_x1_u32r >> 15) * (v_x1_u32r >> 15)) >> 7) *
                               ((int32_t)c->H1)) >> 4));
    
    v_x1_u32r = (v_x1_u32r < 0) ? 0 : v_x1_u32r;
    v_x1_u32r = (v_x1_u32r > 419430400) ? 419430400 : v_x1_u32r;
    
    return (uint32_t)(v_x1_u32r >> 12);
}
```

### 数据读取函数

```c
/**
 * @brief 读取 BME280 数据
 * @param handle: 句柄
 * @param data: 数据输出
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef BME280_ReadData(BME280_Handle_t *handle,
                                   BME280_Data_t *data) {
    uint8_t buffer[8];
    
    // 读取原始数据
    HAL_I2C_Mem_Read(handle->hi2c, handle->address << 1,
                     BME280_REG_PRESSURE,
                     I2C_MEMADD_SIZE_8BIT, buffer, 8, 100);
    
    // 解析原始数据
    int32_t raw_pressure = ((int32_t)buffer[0] << 12) |
                           ((int32_t)buffer[1] << 4) |
                           ((int32_t)buffer[2] >> 4);
    
    int32_t raw_temp = ((int32_t)buffer[3] << 12) |
                       ((int32_t)buffer[4] << 4) |
                       ((int32_t)buffer[5] >> 4);
    
    int32_t raw_humidity = ((int32_t)buffer[6] << 8) |
                           ((int32_t)buffer[7]);
    
    // 补偿计算
    int32_t t_fine = 0;
    int32_t temp = BME280_CompensateTemperature(handle, raw_temp);
    // 注意：需要修改温度函数保存 t_fine
    
    uint32_t pressure = BME280_CompensatePressure(handle, raw_pressure, t_fine);
    uint32_t humidity = BME280_CompensateHumidity(handle, raw_humidity, t_fine);
    
    // 转换为物理量
    data->temperature = temp / 100.0f;
    data->pressure = pressure / 25600.0f;
    data->humidity = humidity / 1024.0f;
    
    // 计算海拔高度 (简化公式)
    data->altitude = 44330.0f * (1.0f - powf(data->pressure / 1013.25f, 0.1903f));
    
    return HAL_OK;
}
```

## ⚡ 低功耗设计

### 功耗模式

| 模式 | 电流 | 说明 |

| 强制 | 3.6μA@1Hz | 单次测量 |

### 强制模式测量

```c
/**
 * @brief 强制模式单次测量
 */
HAL_StatusTypeDef BME280_ForceMeasure(BME280_Handle_t *handle,
                                       BME280_Data_t *data) {
    // 1. 进入强制模式
    uint8_t ctrl_meas = (BME280_OVERSAMPLING_X1 << 5) |
                        (BME280_OVERSAMPLING_X1 << 2) |
                        (BME280_MODE_FORCED << 0);
    HAL_I2C_Mem_Write(handle->hi2c, handle->address << 1,
                      BME280_REG_CTRL_MEAS,
                      I2C_MEMADD_SIZE_8BIT, &ctrl_meas, 1, 100);
    
    // 2. 等待测量完成
    HAL_Delay(10);
    
    // 3. 读取数据
    return BME280_ReadData(handle, data);
}
```

### 常见问题

| 问题 | 解决方案 |
|------|----------|
| 芯片 ID 错误 | 检查 I2C 地址 (0x76 vs 0x77) |

| 气压偏差大 | 校准海平面气压，考虑温度补偿 |
