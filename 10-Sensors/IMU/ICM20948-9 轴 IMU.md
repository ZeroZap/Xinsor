# ICM20948 9 轴 IMU 传感器

> **一句话描述:** InvenSense 旗舰级 9 轴 IMU，集成 3 轴加速度计 +3 轴陀螺仪 +3 轴磁力计，I2C/SPI 双接口，低功耗高性能  
> **标签:** `#sensor/imu` `#sensor/accelerometer` `#sensor/gyroscope` `#sensor/magnetometer` `#protocol/i2c` `#protocol/spi` `#vendor/invensense`  
> **状态:** ✅ 完成

## 🎯 概述

**ICM20948** 是 InvenSense（现 TDK）的旗舰级 9 轴 IMU 传感器，是 MPU6050/MPU9250 的升级换代产品，在精度、功耗和集成度上都有显著提升。

**核心特性:**
- ✨ **9 轴集成:** 3 轴加速度计 + 3 轴陀螺仪 + 3 轴磁力计 (AK09916)
- ✨ **双接口:** 支持 I2C (400kHz) 和 SPI (7MHz)
- ✨ **超低功耗:** 睡眠模式仅 12μA，DMP 模式下 65μA
- ✨ **高精度:** 陀螺仪噪声密度 0.0087°/s/√Hz
- ✨ **嵌入式 DMP:** 硬件姿态解算，支持 PDR (行人航位推算)
- ✨ **宽电压:** 1.71V-3.45V 供电

**典型应用:** VR/AR 设备、高端无人机、机器人导航、可穿戴设备、智能手机、电子罗盘

## 🔌 硬件设计

### 引脚定义 (关键引脚)

| 引脚 | 名称 | 类型 | 功能描述 |
|------|------|------|----------|
| 1 | VDD | P | 电源 (1.71V-3.45V) |
| 2 | GND | P | 电源地 |
| 3 | AD0/CS | I | I2C 地址选择 / SPI 片选 |
| 4 | SCL/SCL | I | I2C 时钟 / SPI 时钟 |
| 5 | SDA/SDO | I/O | I2C 数据 / SPI 数据输出 |
| 6 | SDO/SDI | O/I | I2C 辅助数据 / SPI 数据输入 |
| 7 | INT1 | O | 中断输出 1 |
| 8 | INT2 | O | 中断输出 2 (可选) |
| 9 | VLOGIC | P | 数字 I/O 电源 |
| 10 | FSYNC | I | 帧同步输入 |
| 11 | REGOUT | O | LDO 滤波电容 (10μF) |
| 12 | VDD | P | 电源 |

### I2C vs SPI 配置

```
I2C 模式 (AD0/CS = VDD):
- I2C 地址：0x68 (AD0=GND) 或 0x69 (AD0=VDD)
- 速度：400kHz

SPI 模式 (AD0/CS = GND):
- 4 线制：SCL, SDA(SDO), SDI, CS
- 速度：7MHz
- 模式：Mode 0/3
```

### 典型电路 (I2C 模式)

```
         3.3V
           │
      ┌────┴────┐
      │         │
     ┌┴┐       ┌┴┐
     │ │ 10μF  │ │ 100nF
     │ │       │ │
     └┬┘       └┬┘
      │         │
      ├────┬────┼─── SCL (PB6)
      │    │    │     (4.7kΩ上拉)
      │   GND   ├─── SDA (PB7)
      │         │     (4.7kΩ上拉)
     ┌┴───────────┴┐
     │   ICM20948  │
     │             │
     │ AD0 → GND   │  (地址 0x68)
     │ INT1 → PA0  │  (中断)
     │ REGOUT → 10μF → GND
     └─────────────┘
```

**设计要点:**
1. **REGOUT 电容:** 必须接 10μF 低 ESR 电容，稳定内部 LDO
2. **去耦电容:** 10μF + 100nF 并联，尽量靠近 VDD 引脚
3. **I2C 上拉:** 4.7kΩ (3.3V 系统)，2.2kΩ (高速应用)
4. **PCB 布局:** 远离电机和高频干扰源

## 💻 软件驱动 (STM32 HAL)

### 寄存器定义

```c
#ifndef __ICM20948_H
#define __ICM20948_H

#include "main.h"

// ICM20948 I2C 地址
#define ICM20948_ADDR_0     0x68
#define ICM20948_ADDR_1     0x69

// 用户_bank 寄存器
#define ICM20948_REG_BANK_SEL       0x7F

// Bank 0 - 常用寄存器
#define ICM20948_WHO_AM_I           0x00
#define ICM20948_USER_CTRL          0x03
#define ICM20948_LP_CONFIG          0x05
#define ICM20948_PWR_MGMT_1         0x06
#define ICM20948_PWR_MGMT_2         0x07
#define ICM20948_INT_PIN_CFG        0x0F
#define ICM20948_INT_ENABLE         0x10
#define ICM20948_INT_STATUS         0x17
#define ICM20948_ACCEL_XOUT_H       0x2D
#define ICM20948_GYRO_XOUT_H        0x33
#define ICM20948_TEMP_OUT_H         0x39
#define ICM20948_ACCEL_CONFIG       0x1C
#define ICM20948_GYRO_CONFIG_1      0x1B

// Bank 2 - 磁力计
#define ICM20948_I2C_MST_CTRL       0x24
#define ICM20948_I2C_SLV0_ADDR      0x25
#define ICM20948_I2C_SLV0_REG       0x26
#define ICM20948_I2C_SLV0_CTRL      0x27
#define ICM20948_I2C_SLV0_DO        0x63
#define ICM20948_EXT_SLV_SENS_DATA  0x3B

// Bank 3 - 磁力计寄存器 (AK09916)
#define AK09916_REG_WIA1            0x00
#define AK09916_REG_WIA2            0x01
#define AK09916_REG_ST1             0x02
#define AK09916_REG_HXL             0x03
#define AK09916_REG_CNTL2           0x31
#define AK09916_REG_CNTL3           0x32

// WHO_AM_I 值
#define ICM20948_WHO_AM_I_VALUE     0xEA
#define AK09916_WHO_AM_I_VALUE      0x48

// 数据结构
typedef struct {
    float accel_x, accel_y, accel_z;    // g
    float gyro_x, gyro_y, gyro_z;       // °/s
    float mag_x, mag_y, mag_z;          // μT
    float temperature;                   // °C
} ICM20948_Data_t;

typedef struct {
    I2C_HandleTypeDef *hi2c;
    uint8_t address;
    uint8_t accel_range;
    uint8_t gyro_range;
} ICM20948_Handle_t;

#endif
```

### 银行切换机制

```c
/**
 * @brief 切换寄存器银行
 * @param handle: 句柄
 * @param bank: 银行号 (0-3)
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef ICM20948_SetBank(ICM20948_Handle_t *handle, uint8_t bank) {
    uint8_t bank_sel = (bank << 4) & 0x70;
    return HAL_I2C_Mem_Write(handle->hi2c, handle->address << 1,
                              ICM20948_REG_BANK_SEL,
                              I2C_MEMADD_SIZE_8BIT, &bank_sel, 1, 100);
}
```

### 初始化函数

```c
/**
 * @brief ICM20948 初始化
 * @param handle: 句柄
 * @param hi2c: I2C 句柄
 * @param address: I2C 地址
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef ICM20948_Init(ICM20948_Handle_t *handle,
                                 I2C_HandleTypeDef *hi2c,
                                 uint8_t address) {
    uint8_t chip_id = 0;
    uint8_t temp = 0;
    
    handle->hi2c = hi2c;
    handle->address = address;
    
    // 1. 复位传感器
    temp = 0x80; // BIT_RST
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_PWR_MGMT_1,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    HAL_Delay(100);
    
    // 2. 唤醒 (清除 SLEEP 位)
    temp = 0x01; // CLKSEL
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_PWR_MGMT_1,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    HAL_Delay(10);
    
    // 3. 验证 WHO_AM_I (Bank 0)
    ICM20948_SetBank(handle, 0);
    HAL_I2C_Mem_Read(hi2c, address << 1, ICM20948_WHO_AM_I,
                     I2C_MEMADD_SIZE_8BIT, &chip_id, 1, 100);
    
    if (chip_id != ICM20948_WHO_AM_I_VALUE) {
        return HAL_ERROR;
    }
    
    // 4. 配置加速度计 (±2g)
    temp = 0x00;
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_ACCEL_CONFIG,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    
    // 5. 配置陀螺仪 (±250°/s)
    temp = 0x00;
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_GYRO_CONFIG_1,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    
    // 6. 配置磁力计 (I2C 主模式)
    ICM20948_SetBank(handle, 2);
    
    // 启用 I2C 主模式
    temp = 0x20; // BIT_I2C_MST_EN
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_I2C_MST_CTRL,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    
    // 配置 SLV0 读取磁力计
    temp = 0x8C; // AK09916 地址 0x0C, 读模式
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_I2C_SLV0_ADDR,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    
    temp = AK09916_REG_HXL; // 磁力计数据起始寄存器
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_I2C_SLV0_REG,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    
    temp = 0x86; // 启用 SLV0, 读取 6 字节
    HAL_I2C_Mem_Write(hi2c, address << 1, ICM20948_I2C_SLV0_CTRL,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    
    // 7. 配置磁力计连续测量模式
    ICM20948_SetBank(handle, 3);
    temp = 0x08; // CNTL2: 连续测量 100Hz
    HAL_I2C_Mem_Write(hi2c, address << 1, AK09916_REG_CNTL2,
                      I2C_MEMADD_SIZE_8BIT, &temp, 1, 100);
    
    // 8. 等待磁力计稳定
    HAL_Delay(50);
    
    return HAL_OK;
}
```

### 数据读取函数

```c
/**
 * @brief 读取 9 轴数据
 * @param handle: 句柄
 * @param data: 数据输出
 * @return HAL_StatusTypeDef
 */
HAL_StatusTypeDef ICM20948_ReadData(ICM20948_Handle_t *handle,
                                     ICM20948_Data_t *data) {
    uint8_t buffer[21];
    ICM20948_SetBank(handle, 0);
    
    // 读取 IMU + 磁力计数据 (21 字节)
    HAL_StatusTypeDef status = HAL_I2C_Mem_Read(handle->hi2c,
                                                 handle->address << 1,
                                                 ICM20948_ACCEL_XOUT_H,
                                                 I2C_MEMADD_SIZE_8BIT,
                                                 buffer, 21, 100);
    
    if (status != HAL_OK) return status;
    
    // 解析加速度计 (16-bit, 大端)
    int16_t ax = (buffer[0] << 8) | buffer[1];
    int16_t ay = (buffer[2] << 8) | buffer[3];
    int16_t az = (buffer[4] << 8) | buffer[5];
    
    data->accel_x = ax / 16384.0f;
    data->accel_y = ay / 16384.0f;
    data->accel_z = az / 16384.0f;
    
    // 解析陀螺仪
    int16_t gx = (buffer[6] << 8) | buffer[7];
    int16_t gy = (buffer[8] << 8) | buffer[9];
    int16_t gz = (buffer[10] << 8) | buffer[11];
    
    data->gyro_x = gx / 131.0f;
    data->gyro_y = gy / 131.0f;
    data->gyro_z = gz / 131.0f;
    
    // 解析温度
    int16_t temp = (buffer[12] << 8) | buffer[13];
    data->temperature = temp / 333.87f + 21.0f;
    
    // 解析磁力计 (16-bit, 小端)
    int16_t mx = (buffer[15] << 8) | buffer[14];
    int16_t my = (buffer[17] << 8) | buffer[16];
    int16_t mz = (buffer[19] << 8) | buffer[18];
    
    data->mag_x = mx * 0.15f; // 转换为 mG
    data->mag_y = my * 0.15f;
    data->mag_z = mz * 0.15f;
    
    return HAL_OK;
}
```

## ⚡ 低功耗设计

### 功耗模式

| 模式 | 电流 | 说明 |
|------|------|------|
| 睡眠 | 12μA | 所有传感器关闭 |
| 待机 | 55μA | 加速度计周期唤醒 |
| DMP | 65μA | 硬件姿态解算 |
| 全功能 | 900μA | 9 轴连续测量 |

### 低功耗配置

```c
/**
 * @brief 进入低功耗模式
 */
HAL_StatusTypeDef ICM20948_EnterLowPower(ICM20948_Handle_t *handle) {
    ICM20948_SetBank(handle, 0);
    
    // 配置加速度计周期唤醒 (1.125Hz)
    uint8_t lp_config = 0x20; // BIT_ACCEL_CYCLE
    HAL_I2C_Mem_Write(handle->hi2c, handle->address << 1,
                      ICM20948_LP_CONFIG,
                      I2C_MEMADD_SIZE_8BIT, &lp_config, 1, 100);
    
    // 关闭陀螺仪和磁力计
    uint8_t pwr_mgmt_2 = 0xE0; // 禁用 GYRO 和 MAG
    HAL_I2C_Mem_Write(handle->hi2c, handle->address << 1,
                      ICM20948_PWR_MGMT_2,
                      I2C_MEMADD_SIZE_8BIT, &pwr_mgmt_2, 1, 100);
    
    return HAL_OK;
}
```

## 🔧 调试技巧

### 常见问题

| 问题 | 解决方案 |
|------|----------|
| 磁力计数据不变 | 检查 I2C 主模式配置，确认 SLV0 正确 |
| WHO_AM_I 错误 | 检查银行切换，确保在 Bank 0 读取 |
| 数据跳变 | 增加 DLPF 滤波，检查机械固定 |

### 快速测试

```c

    ICM20948_Data_t data;
    
    printf("=== ICM20948 9-Axis Test ===\r\n");

                   data.gyro_x, data.gyro_y, data.gyro_z);
```
